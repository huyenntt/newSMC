82 void Automaton::printToTransitionGraph(std::vector<std::string> & edgeLabels,
83  util::Output & out,
84  PetriNet::AutomatonConverter converter) const
85 {
86  switch(converter)
87  {
88  case PetriNet::PETRIFY :
89  case PetriNet::GENET : break;
90  default : PNAPI_ASSERT(false); // do not call with other converters
91  }
92 
93  // create and fill stringstream for buffering graph information
94  std::ostringstream TGStringStream; // used as buffer for graph information
95 
96  TGStringStream << ".state graph\n";
97 
98  // copy edges
99  for(unsigned int i = 0; i < edges_.size(); ++i)
100  {
101  Edge * edge = edges_[i];
102 
103  // build label vector:
104  // each label is mapped to his position in edgeLabes
105  int foundPosition = -1;
106  for(int j = 0; j < (int)edgeLabels.size(); ++j)
107  {
108  if (edge->getLabel() == edgeLabels[j])
109  {
110  foundPosition = j;
111  break;
112  }
113  }
114  if (foundPosition == -1)
115  {
116  foundPosition = (int)edgeLabels.size();
117  edgeLabels.push_back(edge->getLabel());
118  }
119 
120  // print current transition to stream
121  TGStringStream << "p" << edge->getSource().getName() << " t" << foundPosition
122  << " p" << edge->getDestination().getName() << "\n";
123  }
124 
125  // mark final states
126  for(unsigned int i = 0; i < states_.size(); ++i)
127  {
128  if (states_[i]->isFinal())
129  {
130  // each label is mapped to his position in edgeLabes
131  std::string currentLabel = "FINAL";
132  currentLabel += states_[i]->getName();
133  int foundPosition = (int)edgeLabels.size();
134  edgeLabels.push_back(currentLabel);
135  TGStringStream << "p" << states_[i]->getName() << " t" << foundPosition << " p00\n";
136  }
137  }
138 
139  TGStringStream << ".marking {p" << (*getInitialStates().begin())->getName() << "}\n";
140  TGStringStream << ".end" << std::flush;
141 
142 
143  // create STG file, print header, transition information and then
144  // add buffered graph information
145  if(converter == PetriNet::PETRIFY)
146  {
147  out.stream() << ".model Labeled_Transition_System\n.dummy";
148  }
149  else
150  {
151  out.stream() << ".outputs";
152  }
153 
154  for (int i = 0; i < (int)edgeLabels.size(); ++i)
155  {
156  out.stream() << " t" << i;
157  }
158 
159  std::string TGGraphString = TGStringStream.str();
160  out.stream() << "\n" << TGGraphString << std::endl;
161 }

175 std::string PetriNet::remap(const std::string & edge, std::vector<std::string> & edgeLabels)
176 {
177  char delimeter = ((automatonConverter_ == PETRIFY) ? '/' : '_');
178 
179  string affix = ((edge.find(delimeter) != string::npos)
180  ? (string("/") + edge.substr(edge.find(delimeter) + 1))
181  : ""); // read affix
182 
183  string indexString = edge.substr(1, edge.size() - affix.size() - 1); // read index
184 
185  istringstream indexStream; // read index as integer
186  int index;
187  indexStream.str(indexString);
188  indexStream >> index;
189 
190  return edgeLabels[index] + affix;
