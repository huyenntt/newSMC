87 bool Node::isParallel(const Node & n2) const
88 {
89  if (
90  (this != &n2) && // precondition 1
91  (getPreset() == n2.getPreset()) && // precondition 2
92  (getPostset() == n2.getPostset()) // precondition 3
93  )
94  {
95  // precondition 4 - preset
96  PNAPI_FOREACH(preArc_, presetArcs_)
97  {
98  if((*preArc_)->getWeight() != net_.findArc(((*preArc_)->getSourceNode()),n2)->getWeight() )
99  return false;
100  }
101 
102  // precondition 4 - postset
103  PNAPI_FOREACH(postArc_, postsetArcs_)
104  {
105  if((*postArc_)->getWeight() != net_.findArc(n2,(*postArc_)->getTargetNode())->getWeight() )
106  return false;
107  }
108 
109  // precondition 5
110  Transition * t1 = const_cast<Transition *>(dynamic_cast<const Transition *>(this));
111  Transition * t2 = const_cast<Transition *>(dynamic_cast<const Transition *>(&n2));
112  if((t1 != NULL) && (t2 != NULL))
113  {
114  return t1->equalLabels(*t2);
115  }
116 
117  // all preconditions fullfilled
118  return true;
119  }
120  return false;
121 }

272 void Node::mergeArcs(const Node & node, bool addWeights, bool isPostset)
273 {
274  const set<Arc *> & arcSet = (isPostset ? (node.postsetArcs_) : (node.presetArcs_));
275  PNAPI_FOREACH(arc2, arcSet)

277  Node & arc1Source = isPostset ? (*this) : ((*arc2)->getSourceNode());
