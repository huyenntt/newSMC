486  const std::string & netPrefix)

489 
490  // mapping from old interface to new interface and new places
491  map<Label *, Label *> label2label;
492  map<Label *, Place *> label2place;
493  // common synchronous labels
494  set<Label *> commonLabels;
495 
496  // compose interface
497  PetriNet result(interface_, net.interface_, label2label, label2place, commonLabels);
498 
499  // ------------ STEP 2 -----------------------------
500 
501  // mapping from source nets' places to result net's places
502  map<const Place *, const Place *> placeMap;
503 
504  // copy internal places of this net
505  PNAPI_FOREACH(p, places_)
506  {
507  placeMap[*p] = &result.createPlace(myPrefix+(*p)->getName(), (*p)->getTokenCount());
508  }
509 
510  // copy internal places of "net"
511  PNAPI_FOREACH(p, net.places_)
512  {
513  placeMap[*p] = &result.createPlace(netPrefix+(*p)->getName(), (*p)->getTokenCount());
514  }
515 
516  // ------------ STEP 3 -----------------------------
517 
518  // sets of transitions to be merged
519  set<Transition *> mergeThis;
520  set<Transition *> mergeOther;
521 
522  // iterate through this net's transitions and copy transitions
523  PNAPI_FOREACH(t, transitions_)
524  {
525  if( (!(*t)->isSynchronized()) || // this transition is not synchronized
526  (util::setIntersection((*t)->getSynchronousLabels(), commonLabels).empty()) ) // transition has no shared label
527  {
528  // create a prefixed transition in the resulting net
529  Transition & rt = result.createTransition(myPrefix+(*t)->getName());
530  rt.setCost((*t)->getCost()); // copy transition costs
531 
532  // copy preset arcs
533  PNAPI_FOREACH(f, (*t)->getPresetArcs())
534  {
535  result.createArc(*const_cast<Place *>(placeMap[static_cast<Place *>(&(*f)->getSourceNode())]), rt, (*f)->getWeight());
536  }
537 
538  // copy postset arcs
539  PNAPI_FOREACH(f, (*t)->getPostsetArcs())
540  {
541  result.createArc(rt, *const_cast<Place *>(placeMap[static_cast<Place *>(&(*f)->getTargetNode())]), (*f)->getWeight());
542  }
543 
544  // copy labels
545  PNAPI_FOREACH(l, (*t)->getLabels())
546  {
547  Label * rl = label2label[l->first]; // label in result net
548  if(rl == NULL) // if no resulting label was found
549  {

551  if(l->first->getType() == Label::INPUT) // if transition was consuming
552  {
553  result.createArc(*label2place[l->first], rt, l->second);
554  }
555  else // transition was producing
556  {

558  }
559  }
560  else

563  }
564  }
565  }
566  else

569  }
570  }

1327 void PetriNet::produce(const PetriNet & net, const std::string & aPrefix,
1328  const std::string & aNetPrefix) throw (exception::InputError)
1329 {
1330  PNAPI_ASSERT(!aPrefix.empty());
1331  PNAPI_ASSERT(!aNetPrefix.empty());
1332  PNAPI_ASSERT(aPrefix != aNetPrefix);
1333  PNAPI_ASSERT(net.interface_.getInputLabels().empty());
1334  PNAPI_ASSERT(net.interface_.getOutputLabels().empty());
1335 
1336  string prefix = aPrefix + ".";
1337  string netPrefix = aNetPrefix + ".";
1338  map<Transition *, set<Transition *> > labels = translateConstraintLabels(net);
1339 
1340  // add places
1341  prefixNodeNames(prefix); // prefix nodes
1342  map<const Place *, const Place *> placeMapping = copyPlaces(net, netPrefix);
1343 
1344  // conjunct final conditions
1345  finalCondition_.conjunct(net.finalCondition_, placeMapping);
1346 
1347  // handle transitions with empty label
1348  PNAPI_FOREACH(it, net.transitions_)
1349  {
1350  Transition & netTrans = **it; // t'
1351  set<Transition *> label;
1352  if (labels.find(&netTrans) != labels.end())
1353  {
1354  label = labels.find(&netTrans)->second;
1355  }
1356  if (label.empty())
1357  {
1358  Transition & trans = createTransition(netTrans, netPrefix);
1359  createArcs(trans, netTrans, &placeMapping);
1360  }
1361  }
1362 
1363  // create product transitions
1364  set<Transition *> labelTransitions;
1365  PNAPI_FOREACH(it1, labels)
1366  {
1367  PNAPI_ASSERT(it1->first != NULL);
1368  Transition & netTrans = *it1->first; // t'
1369  PNAPI_ASSERT(net.containsNode(netTrans));
1370  set<Transition *> ts = it1->second;
1371  PNAPI_FOREACH(it2, ts)
1372  {
1373  PNAPI_ASSERT(*it2 != NULL);
1374  Transition & trans = **it2; // t
1375  PNAPI_ASSERT(containsNode(trans));
1376  labelTransitions.insert(&trans);
1377  Transition & prodTrans = createTransition(); // (t, t')
1378  createArcs(prodTrans, trans);
1379  createArcs(prodTrans, netTrans, &placeMapping);
1380  PNAPI_FOREACH(l, trans.getLabels())
1381  {
1382  prodTrans.addLabel(*(l->first), l->second);
1383  }
1384  }
1385  }
1386 
1387  // remove label transitions
1388  PNAPI_FOREACH(it, labelTransitions)
1389  {
1390  deleteTransition(**it);
1391  }
1392 }

1507 bool PetriNet::isWorkflow() const
1508 {
1509  Place * first = NULL;
1510  Place * last = NULL;
1511 
1512  // each 2 nodes \f$x,y \in P \cup T\f$ are in a directed cycle
1513  // (strongly connected net using tarjan's algorithm)
1514  unsigned int index = 0;
1515  map<Node *, int> indices;
1516  map<Node *, unsigned int> lowlink;
1517  set<Node *> stacked;
1518 
1519  // finding places described in condition (1) & (2)
1520  PNAPI_FOREACH(p, places_)
1521  {
1522  // set all places' index values to ``undefined''
1523  indices[*p] = (-1);
1524 
1525  if((*p)->getPreset().empty())
1526  {
1527  if(first == NULL)
1528  {
1529  first = *p;
1530  }
1531  else
1532  {
1533  return false;
1534  }
1535  }
1536  if((*p)->getPostset().empty())
1537  {
1538  if(last == NULL)
1539  {
1540  last = *p;
1541  }
1542  else
1543  {
1544  return false;
1545  }
1546  }
1547  }
1548 
1549  if((first == NULL) || (last == NULL))
1550  {
1551  return false;
1552  }
1553 
1554  // set all transitions' index values to ``undefined''
1555  PNAPI_FOREACH(t, transitions_)
1556  {
1557  indices[*t] = (-1);
1558  }
1559 
1560  // getting the number of strongly connected components reachable from first
1561  return util::dfsTarjan<Node *>(first, last, stacked, index, indices, lowlink, nodes_.size());
1562 }

1716  set<Label *> inputs = interface_.getInputLabels();
1717  PNAPI_FOREACH(label, inputs)
1718  {
1719  bool needToWrap = false;
1720  // check the postset
1721  PNAPI_FOREACH(t, (*label)->getTransitions())
1722  {
1723  if(!((*t)->isNormal()))
1724  {
1725  needToWrap = true;
1726  break;
1727  }
1728  }
1729 
1730  // if a transition was not normal
1731  if(!needToWrap)
1732  {
1733  continue;
1734  }
1735 
1736  unsigned int complementMarking = 0;
1737  // then we create a wrapper place
1738  Place & np = createPlace();
1739  // and a complement place
1740  Place & cp = createPlace();
1741

1743 
1744  // connect postset of the interface place to the wrapper place
1745  PNAPI_FOREACH(t, transitions)
1746  {
1747  unsigned int weight = (*t)->getLabels().find(*label)->second;
1748  // consume tokens from the wrapper place
1749  createArc(np, **t, weight);
1750  // produce same amount of tokens to the complement place
1751  createArc(**t, cp, weight);
1752  // adjust maximum arc weight
1753  complementMarking = (complementMarking < weight) ? weight : complementMarking;
1754  // remove original "arc"
1755  (*t)->removeLabel(**label);
1756  }
1757 
1758  // connect wrapper place with actual place
1759  Transition & t = createTransition();
1760  t.addLabel(**label);
1761  createArc(t ,np);
1762  createArc(cp, t);
1763 
1764  // adjust complement place
1765  cp.setTokenCount(complementMarking);
1766  // adjust final condition
1767  finalCondition_ = (finalCondition_.getFormula() && (np == 0) && (cp == complementMarking));
1768  }
1769 
1770  // iterate through output places
1771  set<Label *> outputs = interface_.getOutputLabels();
1772  PNAPI_FOREACH(label, outputs)
1773  {
1774  bool needToWrap = false;
1775  // check the preset
1776  PNAPI_FOREACH(t, (*label)->getTransitions())
1777  {
1778  if(!((*t)->isNormal()))
1779  {
1780  needToWrap = true;
1781  break;
1782  }
1783  }
1784

1786  if(!needToWrap)
1787  {
1788  continue;
1789  }
1790 
1791  unsigned int complementMarking = 0;
1792  // then we create a wrapper place
1793  Place & np = createPlace();
1794  // and a complement place
1795  Place & cp = createPlace();
1796 
1797  set<Transition *> transitions = (*label)->getTransitions();
1798  // connect preset of the interface place to the wrapper place
1799  PNAPI_FOREACH(t, transitions)
1800  {
1801  unsigned int weight = (*t)->getLabels().find(*label)->second;
1802  // consume tokens from the wrapper place
1803  createArc(**t, np, weight);
1804  // produce same amount of tokens to the complement place
1805  createArc(cp, **t, weight);
1806  // adjust maximum arc weight
