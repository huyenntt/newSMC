62 /*
63  * \brief modified Tarjan's algorithm.
64  *
65  * \param n Node which is to check
66  * \param last node without postset
67  * \param stacked Set which is needed to identify a node which is already
68  * stacked
69  * \param index the equivalent to Tarjan's index variable
70  * \param indices map which describes the index property of a node
71  * \param lowlink map which describes the lowlink property of a node
72  * \param nodeCount amount of nodes in the net
73  *
74  * \return whether the net has workflow structure
75  *
76  * \note Used by method isWorkflow() to check condition (3) - only
77  * working for this method.
78  *
79  * To Tarjan's algorithm the following modifications have been applied:
80  * 1.: If the recent node is the "last" node, the lowlink will be set to 0,
81  * since the method PetrNet::isWorkflow() actually checks, whether
82  * the net is strongly connected when a single transition, consuming
83  * from the last place and producing to the first place, is added.
84  * So instead of adding this transition, the lowlink value 0 will
85  * directly be assigned.
86  * 2.: If a scc has been found, the index has to be 0 and the stack must
87  * hold nodeCount nodes. Otherwise the net is not strongly connected.
88  */
89 template <typename T>
90 bool dfsTarjan(T n, T last, std::set<T> & stacked,
91  unsigned int & index, std::map<T, int> & indices,
92  std::map<T, unsigned int> & lowlink, unsigned int nodeCount)
93 {
94  indices[n] = lowlink[n] = index++;
95  if(n == last)
96  {
97  lowlink[n] = 0;
98  }
99  stacked.insert(n);
100 
101  PNAPI_FOREACH(nn, n->getPostset())
102  {
103  if(indices[*nn] < 0)
104  {
105  // new node in dfs-tree
106  if(!dfsTarjan(*nn, last, stacked, index, indices, lowlink, nodeCount))
107  {
108  // scc found whithin the net
109  return false;
110  }
111  lowlink[n] = std::min(lowlink[n], lowlink[*nn]);
112  }
113  else
114  {
115  if(stacked.count(*nn) > 0)
116  {
117  lowlink[n] = std::min(lowlink[n], static_cast<unsigned int>(indices[*nn]));
118  }
119  }
120  }
121 
122  if(static_cast<int>(lowlink[n]) == indices[n])
123  {
124  // scc found
125  if((indices[n] == 0) && // this node is the start node
126  (stacked.size() == nodeCount)) // all nodes belong to this scc
127  {
128  return true;
129  }
130  else
131  {
132  return false;
133  }
134  }
135 
136  return true;
137 }
