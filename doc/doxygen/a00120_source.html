105 Output::~Output() {
106  if (&os != &std::cout) {
107  delete(&os);
108  if (temp == NULL) {
109  status("closed file '%s'", filename.c_str());
110  } else {
111  if (keepTempfiles) {
112  status("closed temporary file '%s'", filename.c_str());
113  } else {
114  if (remove(filename.c_str()) == 0) {
115  status("closed and deleted temporary file '%s'", filename.c_str());
116  } else {
117  // this should never happen, because mkstemp creates temp
118  // files in mode 0600.
119  status("closed, but could not delete temporary file '%s'", filename.c_str());
120  }
121  }
122  free(temp);
123  }
124  }
125 }

163 char* Output::createTmp() {
164 #ifdef __MINGW32__
165  temp = basename(const_cast<char*>(tempfileTemplate.c_str()));
166  if (mktemp(temp) == NULL) {
167  throw exception::Error("could not create to temporary file '" + std::string(temp) + "'");
168  };
169 #else
170  temp = strdup(tempfileTemplate.c_str());
171  if (mkstemp(temp) == -1) {
172  throw exception::Error(string("could not create to temporary file '") + temp + "'");
173  };
174 #endif
175  return temp;
176 }
