71 Interface::Interface(PetriNet & net, const Interface & interface1, const Interface & interface2,
72  std::map<Label *, Label *> & label2label, std::map<Label *, Place *> & label2place,
73  std::set<Label *> & commonSynLabels) :
74  net_(net), nextID_(0)
75 {
76  // find common labels
77  set<Label *> commonLabels;
78  {
79  set<string> shapes1, shapes2, commonShapes;
80  PNAPI_FOREACH(p, interface1.getPorts())
81  {
82  set<Label *> labels = p->second->getAllLabels();
83  PNAPI_FOREACH(l, labels)
84  {
85  shapes1.insert((*l)->getName());
86  }
87  }
88  PNAPI_FOREACH(p, interface2.getPorts())
89  {
90  set<Label *> labels = p->second->getAllLabels();
91  PNAPI_FOREACH(l, labels)
92  {
93  shapes2.insert((*l)->getName());
94  }
95  }
96 
97  commonShapes = util::setIntersection(shapes1, shapes2);
98 
99  PNAPI_FOREACH(s, commonShapes)
100  {
101  commonLabels.insert(interface1.findLabel(*s));
102  commonLabels.insert(interface2.findLabel(*s));
103  }
104  }
105 
106  // copy or compose ports of interface1
107  PNAPI_FOREACH(port, interface1.ports_)
108  {
109  set<Label *> labels1 = port->second->getAllLabels();
110  PNAPI_ASSERT_USER(!(labels1.empty()), string("port '") + port->first + "' is empty");
111 
112  if(util::setIntersection(labels1, commonLabels).empty()) // no common labels
113  {
114  // copy port
115  string name = interface1.name_ + port->first;
116  Port * p = new Port(net_, name);
117  ports_[name] = p;
118 
119  PNAPI_FOREACH(label, labels1)
120  {
121  label2label[*label] = &(p->addLabel((*label)->getName(), (*label)->getType()));
122  }
123  }
124  else
125  {
126  // compliance check with composition
127  PNAPI_ASSERT_USER(util::setDifference(labels1, commonLabels).empty(),
128  string("port '") + port->first + "' shares at least one but not all labels",
129  exception::UserCausedError::UE_COMPOSE_ERROR);
130 
131  PNAPI_FOREACH(l1, labels1)
132  {
133  Label * l2 = interface2.findLabel((*l1)->getName());
134  PNAPI_ASSERT_USER((*l1)->getType() + l2->getType() == 4,
135  string("labels '") + l2->getName() + "' do not fit by type",
136  exception::UserCausedError::UE_COMPOSE_ERROR);
137 
138  if(l2->getType() == Label::SYNCHRONOUS)
139  {
140  commonSynLabels.insert(*l1);
141  commonSynLabels.insert(l2);
142  }
143  else
144  {
145  // replace label by a place
146  Place * p = &(net_.createPlace(l2->getName()));
147  label2place[*l1] = p;
148  label2place[l2] = p;
149  }
150  }
151  }
152  }
153 
154  // copy remaining ports of interface2
155  PNAPI_FOREACH(port, interface2.ports_)
156  {
157  set<Label *> labels2 = port->second->getAllLabels();
158  if(util::setIntersection(labels2, commonLabels).empty()) // no common label
159  {
160  // copy port
161  string name = interface2.name_ + port->first;
162  Port * p = new Port(net_, name);
163  ports_[name] = p;
164

166  {

168  }

174  string("port '") + port->first + "' shares at least one but not all labels",

241 Port & Interface::mergePorts(Port & port1, Port & port2, const std::string & name)
242 {
243  map<string, Label::Type> types;
244  map<string, map<Transition *, unsigned int> > arcs;
245 
246  set<Label *> labels = port1.getAllLabels();
247 
248  PNAPI_FOREACH(l, labels)
249  {
250  types[(*l)->getName()] = (*l)->getType();
251  PNAPI_FOREACH(t, (*l)->getTransitions())
252  {
253  arcs[(*l)->getName()][*t] = (*t)->getLabels().find(*l)->second;
254  }
255  }
256 
257  labels = port2.getAllLabels();
258 
259  PNAPI_FOREACH(l, labels)
260  {
261  types[(*l)->getName()] = (*l)->getType();
262  PNAPI_FOREACH(t, (*l)->getTransitions())
263  {
264  arcs[(*l)->getName()][*t] = (*t)->getLabels().find(*l)->second;
265  }
266  }
267 
268  string newName = ((name == "") ? port1.getName() : name);
269 
270  ports_.erase(port1.getName());
271  ports_.erase(port2.getName());
272  delete (&port1);
273  delete (&port2);
274 
275  Port * result = new Port(net_, newName);
276  ports_[newName] = result;
277 
278  PNAPI_FOREACH(t, types)
279  {
280  Label & l = result->addLabel(t->first, t->second);
281  map<Transition *, unsigned int> & arcs_ = arcs[t->first];
282 
283  PNAPI_FOREACH(a, arcs_)
284  {
285  a->first->addLabel(l, a->second);
286  }
287  }
288 
289  return *result;
290 }
